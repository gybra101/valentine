            0%, 100% { transform: translateY(0); }

            display: none;

        }

        @keyframes float {

            to { transform: translateY(-100px); opacity: 0; }

    <h1>Happy Valentine's Day!</h1>

    <p>Click the heart for a special message</p>

    <div class="heart" onclick="showMessage()">❤️</div>

    <p class="message" id="valentineMessage"></p>

    

    <script>

        function showMessage() {

            document.querySelector('.heart').style.animation = 'none';

            document.getElementById('valentineMessage').innerText = "You must be an API request... because you just returned my heart! ❤️";

            document.getElementById('valentineMessage').style.display = 'block';

            createFloatingHearts();

        }

        

        function createFloatingHearts() {

            for (let i = 0; i < 10; i++) {

                let heart = document.createElement('div');

                heart.classList.add('floating-heart');

                heart.innerText = '❤️';

                heart.style.left = Math.random() * window.innerWidth + 'px';

                heart.style.top = Math.random() * window.innerHeight + 'px';

                document.body.appendChild(heart);

                setTimeout(() => heart.remove(), 2000);

            }

        }

    </script>
